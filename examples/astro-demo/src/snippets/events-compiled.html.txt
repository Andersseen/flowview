<button
  data-flow-on-click="save"
  data-flow-scope="a1b2c3d4e5f6"
  data-flow-args='[{"__flow":"$event"}]'
>
  Save
</button>

<script>
  function save(event) {
    const output = document.querySelector("#event-output");
    output.textContent = `Saved from a ${event.type} event`;
  }

  function removeItem(id, element) {
    element.setAttribute("disabled", "true");
  }

  import { registerFlowHandlers } from "@flowview/dom/runtime";
  registerFlowHandlers("a1b2c3d4e5f6", { removeItem, save }, ["click"]);
</script>
